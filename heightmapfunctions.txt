using System;
using System.Collections.Generic;
using VoxelEngine_Silk.Net_1._0.World;

namespace VoxelEngine_Silk.Net_1._0.World;

public enum BiomeType { Desert, Plains, Forest, Tundra, Mountains, Ocean }

public static class BiomeManager
{
    public const float SEA_LEVEL = 62f;

    public static void InitializeNoise(VoxelWorld world, int seed)
    {
        // 1. Continentalness: Large scale land/sea (Terrain First)
        world.ContinentalNoise.SetSeed(seed);
        world.ContinentalNoise.SetFrequency(0.0002f); 
        world.ContinentalNoise.SetNoiseType(FastNoiseLite.NoiseType.OpenSimplex2);
        world.ContinentalNoise.SetFractalType(FastNoiseLite.FractalType.FBm);
        world.ContinentalNoise.SetFractalOctaves(4);

        // 2. Erosion: Used to create slopes and "ruffle" the heightmap
        world.ErosionNoise.SetSeed(seed + 10);
        world.ErosionNoise.SetFrequency(0.001f);
        world.ErosionNoise.SetFractalType(FastNoiseLite.FractalType.FBm);

        // 3. Peaks & Valleys: Independent mountain ranges
        world.RiverNoise.SetSeed(seed + 4);
        world.RiverNoise.SetFrequency(0.0006f);
        world.RiverNoise.SetNoiseType(FastNoiseLite.NoiseType.OpenSimplex2);
        world.RiverNoise.SetFractalType(FastNoiseLite.FractalType.Ridged);

        // 4. Climate: Only used as a lookup "label" after terrain is shaped
        world.TempNoise.SetSeed(seed + 3);
        world.TempNoise.SetFrequency(0.0001f); 
        world.HumidityNoise.SetSeed(seed + 2);
        world.HumidityNoise.SetFrequency(0.0001f);

        PlainsBiome.Initialize(seed);
        ForestBiome.Initialize(seed);
        MountainsBiome.Initialize(seed);
    }

    public static float GetHeightAt(VoxelWorld world, float wx, float wz)
    {
        // STEP 1: Generate a global heightmap (Terrain First)
        float continentalness = world.ContinentalNoise.GetNoise(wx, wz);
        float erosion = world.ErosionNoise.GetNoise(wx, wz);
        float pv = world.RiverNoise.GetNoise(wx, wz);

        // Calculate base height based on Continentalness
        float height = SEA_LEVEL;

        if (continentalness < -0.1f) 
        {
            // Ocean Depths
            height += (continentalness + 0.1f) * 30f;
        }
        else 
        {
            // Land: Start with a base elevation
            height += 4f;

            // Add Mountain Ranges (Terrain shaping independent of biome)
            if (pv > 0.5f && continentalness > 0.05f)
            {
                float mountainWeight = Math.Clamp((pv - 0.5f) * 2f, 0, 1);
                height += MountainsBiome.GetHeight(wx, wz, pv) * mountainWeight;
            }

            // Add rolling hills from Erosion
            height += erosion * 10f;
        }

        return height;
    }

    public static BiomeType GetBiomeAt(VoxelWorld world, float wx, float wz)
    {
        // STEP 2: Determine Biome based on the existing terrain (Biome Second)
        float continentalness = world.ContinentalNoise.GetNoise(wx, wz);
        
        // If it's underwater, it's an ocean. Period.
        if (continentalness < -0.1f) return BiomeType.Ocean;

        // Sample climate
        float t = (world.TempNoise.GetNoise(wx, wz) + 1f) / 2f;
        float h = (world.HumidityNoise.GetNoise(wx, wz) + 1f) / 2f;
        float pv = world.RiverNoise.GetNoise(wx, wz);

        // If the terrain is already very high, label it a mountain
        if (pv > 0.7f && continentalness > 0.1f) return BiomeType.Mountains;

        // Otherwise, classify based on climate matrix (Minecraft style)
        if (t < 0.4f) return BiomeType.Tundra;
        if (t > 0.7f && h < 0.4f) return BiomeType.Desert;
        if (h > 0.6f) return BiomeType.Forest;

        return BiomeType.Plains;
    }

    public static byte GetSurfaceBlock(BiomeType type, float wx, float wz)
    {
        return type switch
        {
            BiomeType.Mountains => 5, // Stone
            BiomeType.Forest => ForestBiome.GetForestSurfaceBlock(wx, wz),
            BiomeType.Desert => 4,    // Sand
            BiomeType.Tundra => 11,   // Snow
            _ => 1                    // Grass
        };
    }

    public static byte GetFillerBlock(BiomeType type) => 2;

    public static bool IsLocalWater(BiomeType type, float wx, float wz) => type == BiomeType.Ocean;

    
}
